# Gen 8+ Pokedex Support — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add full Pokedex support for Sword/Shield, Brilliant Diamond/Shining Pearl, Legends Arceus, Scarlet/Violet, and Legends Z-A — with manual entry covering Tera types, alpha pokemon, Mega Evolutions, regional forms, and the Gen 8 marks system.

**Architecture:** Reference data is fetched from PokeAPI + Bulbapedia and committed as JSON; seeder ingests into existing/new SQLite tables; backend exposes new `/api/reference/*` endpoints; frontend gets a progressive-disclosure manual entry form, two new dex lenses (Tera, Alpha), and filter pills for the new attributes.

**Tech Stack:** Bun + Express 5 + bun:sqlite, React 19 + TS + Vite + Tailwind v4, PokeAPI, Bulbapedia (via existing `mediawiki.ts` service).

**Verification model:** This project has no unit-test framework (CLAUDE.md: "TypeScript + ESLint are the only checks"). Each task includes type-check, lint, and a manual smoke step in lieu of automated tests.

---

## Spec deviations from the design doc

Two adjustments after scouting the existing schema (resolve before implementing):

1. **`species_marks_availability` table dropped (YAGNI).** Existing `marks.games` JSON column already records which games each mark is available in; per-species mark availability is not needed for the filter use case. The ManualEntryForm filters marks by `origin_game` only.
2. **`species_forms.is_mega` column not added.** The schema already uses `form_category` for this distinction (`'standard' | 'mega' | 'gigantamax' | 'regional' | ...`). Z-A megas are seeded as `form_category = 'mega'` with `is_collectible = 1`. The `is_mega` flag on `collection_manual` is added (per-instance — "this user's catch is in Mega state"); on `species_forms` we filter by `form_category`.

```typescript
/**
 * Fetches per-game Pokédex membership for Gen 8/9 from PokeAPI's /pokedex
 * endpoint. Output: { [game]: { [dex_name]: [{ species_id, dex_number }] } }.
 *
 * Run: bun run server/src/scripts/fetch-gen8-9-reference.ts
 */
import { writeFileSync } from 'node:fs';
import { join } from 'node:path';

const POKEAPI = 'https://pokeapi.co/api/v2';

// Map game → list of pokedexes available in that game.
// Pokédex names match PokeAPI's `name` field (slug form).
const GAME_DEXES: Record<string, string[]> = {
  sword:             ['galar', 'isle-of-armor', 'crown-tundra'],
  shield:            ['galar', 'isle-of-armor', 'crown-tundra'],
  'brilliant-diamond': ['updated-sinnoh'],
  'shining-pearl':     ['updated-sinnoh'],
  'legends-arceus':  ['hisui'],
  scarlet:           ['paldea', 'kitakami', 'blueberry'],
  violet:            ['paldea', 'kitakami', 'blueberry'],
  // Z-A pokedex(es) are filled in by fetch-za-data.ts; left empty here on
  // purpose so re-running this script doesn't clobber Z-A data.
};

interface DexEntry { species_id: number; dex_number: number; }

async function fetchDex(name: string): Promise<DexEntry[]> {
  const r = await fetch(`${POKEAPI}/pokedex/${name}`);
  if (!r.ok) throw new Error(`PokeAPI /pokedex/${name} → ${r.status}`);
  const j = await r.json() as any;
  return j.pokemon_entries.map((e: any) => ({
    species_id: Number(e.pokemon_species.url.match(/\/pokemon-species\/(\d+)\//)?.[1]),
    dex_number: e.entry_number,
  })).filter((e: DexEntry) => Number.isFinite(e.species_id));
}

async function main() {
  const out: Record<string, Record<string, DexEntry[]>> = {};
  const seen = new Map<string, DexEntry[]>();
  for (const [game, dexes] of Object.entries(GAME_DEXES)) {
    out[game] = {};
    for (const dex of dexes) {
      let entries = seen.get(dex);
      if (!entries) {
        console.log(`  fetching pokedex/${dex}`);
        entries = await fetchDex(dex);
        seen.set(dex, entries);
      }
      out[game][dex] = entries;
    }
  }
  const path = join(import.meta.dir, '..', 'seeds', 'data', 'gen8-9-reference.json');
  writeFileSync(path, JSON.stringify(out, null, 2) + '\n');
  console.log(`\nWrote ${path}`);
  for (const [game, dexes] of Object.entries(out)) {
    const total = Object.values(dexes).reduce((n, e) => n + e.length, 0);
    console.log(`  ${game}: ${total} entries across ${Object.keys(dexes).length} dexes`);
  }
}

await main();
```

```typescript
/**
 * Fetches the complete marks catalog from Bulbapedia ("List of marks") and
 * writes it to seeds/data/marks-catalog.json. Subsequent seed run merges this
 * with the existing `marks` table — keeping IDs stable for marks already
 * present, inserting new ones with auto-incremented IDs.
 *
 * Run: bun run server/src/scripts/fetch-marks-catalog.ts
 */
import { writeFileSync } from 'node:fs';
import { join } from 'node:path';
import { getPageSections, getSectionWikitext } from '../services/mediawiki.js';

interface MarkEntry {
  name: string;          // 'Lunchtime Mark'
  category: string;      // 'Time' | 'Weather' | 'Personality' | 'Rare' | 'Mightiest'
  title_suffix: string;  // 'the Peckish'
  games: string[];       // ['sword','shield','scarlet','violet']
  how_to_obtain: string;
}

async function main() {
  const sections = await getPageSections('List of marks');
  const out: MarkEntry[] = [];

  for (const sec of sections) {
    // Categories on the wiki are headed by section anchors like
    // "Personality marks", "Time marks", "Weather marks", "Rare marks", "Mightiest marks".
    const m = sec.line.match(/^(Personality|Time|Weather|Rare|Mightiest) marks?$/i);
    if (!m) continue;
    const category = m[1];
    const wt = await getSectionWikitext('List of marks', sec.index);

    // Each mark row in the wikitable looks like:
    //   |[[File:...png|24px]]||{{m|Lunchtime Mark}}||"Peckish"||{{game|Sword}} {{game|Shield}} ...||A Pokémon...
    const rows = wt.split(/\n\|-/).slice(1);
    for (const row of rows) {
      const nameMatch = row.match(/\{\{m\|([^|}]+)\}\}/) || row.match(/\|\s*([A-Z][a-zA-Z' ]+ Mark)\s*\|/);
      const titleMatch = row.match(/"([^"]+)"/);
      const games: string[] = [];
      const gameMatches = row.matchAll(/\{\{game\|([^}|]+)/g);
      for (const g of gameMatches) games.push(g[1].toLowerCase().replace(/\s+/g, '-'));
      const howMatch = row.match(/\|\s*([A-Z][^|}\n]{20,})\s*$/m);

      if (!nameMatch) continue;
      out.push({
        name: nameMatch[1].trim(),
        category,
        title_suffix: titleMatch?.[1] ?? '',
        games,
        how_to_obtain: howMatch?.[1]?.trim() ?? '',
      });
    }
  }

  const path = join(import.meta.dir, '..', 'seeds', 'data', 'marks-catalog.json');
  writeFileSync(path, JSON.stringify(out, null, 2) + '\n');
  console.log(`Wrote ${out.length} marks to ${path}`);

  // Sanity: per-category counts
  const byCat = out.reduce<Record<string, number>>((acc, m) => {
    acc[m.category] = (acc[m.category] ?? 0) + 1;
    return acc;
  }, {});
  for (const [cat, n] of Object.entries(byCat)) console.log(`  ${cat}: ${n}`);
}

await main();
```

```typescript
/**
 * Fetches Legends Z-A specific data from Bulbapedia:
 *   - Lumiose / Z-A regional dex.
 *   - New Mega Evolutions introduced or returning in Z-A.
 *   - Origin-mark image (saved to client/src/assets/sprites/origin-marks/lumiose.png).
 *
 * Output: server/src/seeds/data/legends-za.json with shape:
 *   { dex: [{ species_id, dex_number, name }],
 *     new_megas: [{ species_id, name, form_name }],
 *     origin_mark: { source_url, saved: boolean } }
 *
 * If a target page can't be fetched, the script logs a clear error and exits
 * non-zero — we want loud failure so the seeder doesn't ship empty Z-A data.
 *
 * Run: bun run server/src/scripts/fetch-za-data.ts
 */
import { writeFileSync, mkdirSync } from 'node:fs';
import { join, dirname } from 'node:path';
import { getPageSections, getSectionWikitext } from '../services/mediawiki.js';

const ROOT = join(import.meta.dir, '..', '..', '..');
const OUT_JSON = join(import.meta.dir, '..', 'seeds', 'data', 'legends-za.json');
const OUT_MARK = join(ROOT, 'client', 'src', 'assets', 'sprites', 'origin-marks', 'lumiose.png');

interface DexEntry { species_id: number; dex_number: number; name: string; }
interface MegaEntry { species_id: number; name: string; form_name: string; }

async function fetchDex(): Promise<DexEntry[]> {
  // Bulbapedia page name may be one of:
  //   "Lumiose Pokédex"  | "Pokémon Legends: Z-A regional Pokédex" | "Z-A Pokédex"
  // Try in order.
  for (const candidate of ['Lumiose Pokédex', 'Pokémon Legends: Z-A regional Pokédex', 'Z-A Pokédex']) {
    try {
      const sections = await getPageSections(candidate);
      const target = sections.find(s => /list|pokedex/i.test(s.line));
      if (!target) continue;
      const wt = await getSectionWikitext(candidate, target.index);
      // Rows: |001||...||{{p|Chespin}}|| ...
      const rows = wt.split(/\n\|-/).slice(1);
      const out: DexEntry[] = [];
      for (const row of rows) {
        const num = row.match(/\|(\d{3,4})/)?.[1];
        const name = row.match(/\{\{p\|([^|}]+)\}\}/)?.[1];
        if (!num || !name) continue;
        // Map name → species_id via PokeAPI.
        const r = await fetch(`https://pokeapi.co/api/v2/pokemon-species/${name.toLowerCase().replace(/\s+/g, '-')}`);
        if (!r.ok) continue;
        const j = await r.json() as any;
        out.push({ species_id: j.id, dex_number: parseInt(num, 10), name });
      }
      if (out.length > 0) return out;
    } catch { /* try next */ }
  }
  throw new Error('Could not locate the Z-A regional Pokédex page on Bulbapedia');
}

async function fetchNewMegas(): Promise<MegaEntry[]> {
  // Bulbapedia article: "Mega Evolution" or "List of Pokémon by Mega Evolution"
  const candidate = 'Mega Evolution';
  const sections = await getPageSections(candidate);
  const za = sections.find(s => /Z-?A|Legends Z/i.test(s.line));
  if (!za) {
    console.warn(`[za] No Z-A section on "${candidate}"; new megas list is empty`);
    return [];
  }
  const wt = await getSectionWikitext(candidate, za.index);
  const out: MegaEntry[] = [];
  // Rows can be "{{p|Chesnaught}}" with a Mega form.
  const matches = wt.matchAll(/\{\{p\|([A-Za-z' -]+)\}\}/g);
  for (const m of matches) {
    const name = m[1];
    const r = await fetch(`https://pokeapi.co/api/v2/pokemon-species/${name.toLowerCase().replace(/\s+/g, '-')}`);
    if (!r.ok) continue;
    const j = await r.json() as any;
    out.push({ species_id: j.id, name, form_name: `Mega ${name}` });
  }
  // De-dup by species_id (the section may reference each species multiple times).
  const seen = new Set<number>();
  return out.filter(e => seen.has(e.species_id) ? false : (seen.add(e.species_id), true));
}

async function fetchOriginMark(): Promise<{ source_url: string; saved: boolean }> {
  // Bulbapedia "Origin mark" article lists the Lumiose mark image filename.
  const sections = await getPageSections('Origin mark');
  const lumiose = sections.find(s => /Lumiose|Legends Z/i.test(s.line));
  if (!lumiose) return { source_url: '', saved: false };
  const wt = await getSectionWikitext('Origin mark', lumiose.index);
  const img = wt.match(/\[\[File:(Origin mark[^|\]]*)\]?\]?/i)?.[1] ?? wt.match(/\[\[File:([^|\]]+\.png)/)?.[1];
  if (!img) return { source_url: '', saved: false };
  // Use Bulbapedia's Special:Filepath to get the actual binary URL.
  const url = `https://bulbapedia.bulbagarden.net/wiki/Special:Filepath/${encodeURIComponent(img)}`;
  const r = await fetch(url);
  if (!r.ok) {
    console.warn(`[za] origin-mark fetch failed (${r.status}); skipping image save`);
    return { source_url: url, saved: false };
  }
  mkdirSync(dirname(OUT_MARK), { recursive: true });
  const buf = new Uint8Array(await r.arrayBuffer());
  writeFileSync(OUT_MARK, buf);
  return { source_url: url, saved: true };
}

async function main() {
  console.log('[za] fetching regional dex…');
  const dex = await fetchDex();
  console.log(`[za] dex: ${dex.length} entries`);

  console.log('[za] fetching new megas…');
  const new_megas = await fetchNewMegas();
  console.log(`[za] new megas: ${new_megas.length} entries`);

  console.log('[za] fetching origin mark image…');
  const origin_mark = await fetchOriginMark();
  console.log(`[za] origin mark: ${origin_mark.saved ? 'saved' : 'NOT saved (will render no badge)'}`);

  writeFileSync(OUT_JSON, JSON.stringify({ dex, new_megas, origin_mark }, null, 2) + '\n');
  console.log(`Wrote ${OUT_JSON}`);
}

await main();
```

```typescript
/**
 * Idempotent UPSERT for collection_legs. The original `seedCollectionLegs()`
 * early-returns if any rows exist — fine for first-run seeding, but doesn't
 * propagate JSON edits (e.g. flipping 'upcoming' → 'active', adding a new
 * leg). This runs after the initial seed and reconciles JSON ↔ DB.
 */
import { readFileSync } from 'fs';
import { join } from 'path';
import db from '../db.js';
import { paths } from '../paths.js';

interface LegJson {
  key: string;
  label: string;
  origin_mark: string | null;
  games: string[];
  leg_order: number;
  status: string;
}

export function upsertCollectionLegs(): void {
  const data: LegJson[] = JSON.parse(
    readFileSync(join(paths.seedDataDir, 'meta/collection-legs.json'), 'utf-8'),
  );

  const upsert = db.prepare(`
    INSERT INTO collection_legs (key, label, origin_mark, games, leg_order, status)
    VALUES (?, ?, ?, ?, ?, ?)
    ON CONFLICT(key) DO UPDATE SET
      label       = excluded.label,
      origin_mark = excluded.origin_mark,
      games       = excluded.games,
      leg_order   = excluded.leg_order,
      status      = excluded.status
  `);

  const tx = db.transaction(() => {
    for (const leg of data) {
      upsert.run(
        leg.key,
        leg.label,
        leg.origin_mark,
        JSON.stringify(leg.games),
        leg.leg_order,
        leg.status,
      );
    }
  });
  tx();
  console.log(`  Upserted ${data.length} collection legs.`);
}
```

```typescript
/**
 * Seeds Gen 8/9 reference tables from the JSON files produced by
 * fetch-gen8-9-reference.ts, fetch-marks-catalog.ts, fetch-za-data.ts, and
 * the hand-written tera-types.json / paradox-species.json / alpha-species.json.
 *
 * Idempotent: every INSERT uses INSERT OR REPLACE / ON CONFLICT semantics.
 */
import { readFileSync, existsSync } from 'fs';
import { join } from 'path';
import db from '../db.js';
import { paths } from '../paths.js';

function loadJson<T>(rel: string): T | null {
  const path = join(paths.seedDataDir, rel);
  if (!existsSync(path)) {
    console.warn(`  [gen8-9] missing ${rel}; skipping`);
    return null;
  }
  return JSON.parse(readFileSync(path, 'utf-8')) as T;
}

export function seedGen89Reference(): void {
  // 1. Tera types
  const teraTypes = loadJson<Array<{ key: string; name: string; color: string }>>('meta/tera-types.json');
  if (teraTypes) {
    const stmt = db.prepare(`INSERT OR REPLACE INTO tera_types_catalog (key, name, color) VALUES (?, ?, ?)`);
    db.transaction(() => {
      for (const t of teraTypes) stmt.run(t.key, t.name, t.color);
    })();
    console.log(`  Seeded ${teraTypes.length} tera types.`);
  }

  // 2. species_in_dex from gen8-9-reference.json + legends-za.json
  const ref = loadJson<Record<string, Record<string, Array<{ species_id: number; dex_number: number }>>>>('gen8-9-reference.json');
  const za = loadJson<{ dex: Array<{ species_id: number; dex_number: number; name: string }> }>('legends-za.json');
  const insertDex = db.prepare(`
    INSERT OR REPLACE INTO species_in_dex (species_id, game, dex_name, dex_number) VALUES (?, ?, ?, ?)
  `);
  let dexRows = 0;
  db.transaction(() => {
    if (ref) {
      for (const [game, dexes] of Object.entries(ref)) {
        for (const [dexName, entries] of Object.entries(dexes)) {
          for (const e of entries) {
            insertDex.run(e.species_id, game, dexName, e.dex_number);
            dexRows++;
          }
        }
      }
    }
    if (za && za.dex) {
      for (const e of za.dex) {
        insertDex.run(e.species_id, 'legends-z-a', 'lumiose', e.dex_number);
        dexRows++;
      }
    }
  })();
  console.log(`  Seeded ${dexRows} species_in_dex rows.`);

  // 3. Marks catalog merge — only insert marks NOT already in `marks` (by name).
  const marksCatalog = loadJson<Array<{ name: string; category: string; title_suffix: string; games: string[]; how_to_obtain: string }>>('marks-catalog.json');
  if (marksCatalog) {
    const existing = new Set((db.prepare('SELECT name FROM marks').all() as Array<{ name: string }>).map(r => r.name));
    const insertMark = db.prepare(`
      INSERT INTO marks (name, category, how_to_obtain, games, title_suffix, sprite_key, sort_order)
      VALUES (?, ?, ?, ?, ?, ?, (SELECT COALESCE(MAX(sort_order), 0) + 1 FROM marks))
    `);
    let added = 0;
    db.transaction(() => {
      for (const m of marksCatalog) {
        if (existing.has(m.name)) continue;
        insertMark.run(m.name, m.category, m.how_to_obtain, JSON.stringify(m.games), m.title_suffix, '');
        added++;
      }
    })();
    console.log(`  Marks catalog: ${added} new entries added (${existing.size} already present).`);
  }

  // 4. Z-A new megas → species_forms
  if (za && (za as any).new_megas) {
    const insertForm = db.prepare(`
      INSERT OR IGNORE INTO species_forms (species_id, form_name, form_order, form_category, is_battle_only, is_collectible)
      VALUES (?, ?, (SELECT COALESCE(MAX(form_order), 0) + 1 FROM species_forms WHERE species_id = ?), 'mega', 0, 1)
    `);
    let added = 0;
    db.transaction(() => {
      for (const m of (za as any).new_megas as Array<{ species_id: number; form_name: string }>) {
        const r = insertForm.run(m.species_id, m.form_name, m.species_id);
        if (Number(r.changes) > 0) added++;
      }
    })();
    console.log(`  Z-A megas: ${added} new species_forms rows added.`);
  }
}
```

```typescript
import { useEffect, useState } from 'react';

type GameKey = string;

interface FormState {
  species_id: number | '';
  origin_game: GameKey | '';
  is_shiny: boolean;
  // Identity
  form_id?: number;
  gender?: 'M' | 'F' | 'N';
  nickname?: string;
  // Catch
  caught_date?: string;
  caught_location?: string;
  ball?: string;
  level?: number;
  ot_name?: string;
  ot_tid?: number;
  ot_sid?: number;
  // Battle
  nature?: string;
  ability?: string;
  is_hidden_ability?: boolean;
  held_item?: string;
  ivs?: { hp: number; atk: number; def: number; spa: number; spd: number; spe: number };
  evs?: { hp: number; atk: number; def: number; spa: number; spd: number; spe: number };
  moves?: Array<{ name: string; pp_ups: number }>;
  // Cosmetic / game-specific
  ribbons?: string[];
  marks?: string[];
  tera_type?: string;
  is_alpha?: boolean;
  is_mega?: boolean;
  notes?: string;
}

interface Props {
  onSubmit: (state: FormState) => Promise<void>;
  onCancel: () => void;
  activeLegs: Array<{ key: string; label: string; games: string[] }>;
}

const GEN_OF: Record<string, number> = {
  red: 1, blue: 1, yellow: 1,
  gold: 2, silver: 2, crystal: 2,
  ruby: 3, sapphire: 3, emerald: 3, firered: 3, leafgreen: 3,
  diamond: 4, pearl: 4, platinum: 4, heartgold: 4, soulsilver: 4,
  black: 5, white: 5, black2: 5, white2: 5,
  x: 6, y: 6, 'omega-ruby': 6, 'alpha-sapphire': 6,
  sun: 7, moon: 7, 'ultra-sun': 7, 'ultra-moon': 7, 'lets-go-pikachu': 7, 'lets-go-eevee': 7,
  sword: 8, shield: 8, 'brilliant-diamond': 8, 'shining-pearl': 8, 'legends-arceus': 8,
  scarlet: 9, violet: 9, 'legends-z-a': 9,
};

export function ManualEntryForm({ onSubmit, onCancel, activeLegs }: Props) {
  const [state, setState] = useState<FormState>({ species_id: '', origin_game: '', is_shiny: false });
  const [openGroups, setOpenGroups] = useState({ identity: true, catch: false, battle: false, cosmetic: false });
  const [teraTypes, setTeraTypes] = useState<Array<{ key: string; name: string; color: string }>>([]);
  const [marks, setMarks] = useState<Array<{ name: string; category: string }>>([]);
  const [submitting, setSubmitting] = useState(false);

  // Fetch tera types once.
  useEffect(() => {
    fetch('/api/reference/tera-types').then(r => r.json()).then(setTeraTypes).catch(() => {});
  }, []);

  // Refetch marks whenever origin_game changes.
  useEffect(() => {
    if (!state.origin_game) { setMarks([]); return; }
    fetch(`/api/reference/marks?game=${encodeURIComponent(state.origin_game)}`)
      .then(r => r.json()).then(setMarks).catch(() => setMarks([]));
  }, [state.origin_game]);

  const gen = state.origin_game ? GEN_OF[state.origin_game] ?? 0 : 0;
  const showTera     = gen >= 9;
  const showAlpha    = state.origin_game === 'legends-arceus';
  const showIsMega   = state.origin_game === 'legends-z-a';
  const showRibbons  = gen >= 3;
  const showMarks    = gen >= 8;

  const games = activeLegs.flatMap(l => l.games);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    if (!state.species_id || !state.origin_game) return;
    setSubmitting(true);
    try { await onSubmit(state); } finally { setSubmitting(false); }
  };

  return (
    <form onSubmit={handleSubmit} className="space-y-4 p-4">
      {/* Identity */}
      <Group title="Identity" open={openGroups.identity} onToggle={() => setOpenGroups(g => ({ ...g, identity: !g.identity }))}>
        <input type="number" placeholder="Species ID" value={state.species_id}
               onChange={e => setState(s => ({ ...s, species_id: e.target.value ? Number(e.target.value) : '' }))}
               required />
        <select value={state.origin_game} onChange={e => setState(s => ({ ...s, origin_game: e.target.value }))} required>
          <option value="">Origin game…</option>
          {games.map(g => <option key={g} value={g}>{g}</option>)}
        </select>
        <label><input type="checkbox" checked={state.is_shiny} onChange={e => setState(s => ({ ...s, is_shiny: e.target.checked }))} /> Shiny</label>
        <select value={state.gender ?? ''} onChange={e => setState(s => ({ ...s, gender: (e.target.value || undefined) as any }))}>
          <option value="">Gender…</option><option value="M">M</option><option value="F">F</option><option value="N">—</option>
        </select>
        <input type="text" placeholder="Nickname" value={state.nickname ?? ''} onChange={e => setState(s => ({ ...s, nickname: e.target.value }))} />
      </Group>

      {/* Catch */}
      <Group title="Catch" open={openGroups.catch} onToggle={() => setOpenGroups(g => ({ ...g, catch: !g.catch }))}>
        <input type="date" value={state.caught_date ?? ''} onChange={e => setState(s => ({ ...s, caught_date: e.target.value }))} />
        <input type="text" placeholder="Caught location" value={state.caught_location ?? ''} onChange={e => setState(s => ({ ...s, caught_location: e.target.value }))} />
        <input type="text" placeholder="Ball" value={state.ball ?? ''} onChange={e => setState(s => ({ ...s, ball: e.target.value }))} />
        <input type="number" placeholder="Level" min={1} max={100} value={state.level ?? ''} onChange={e => setState(s => ({ ...s, level: e.target.value ? Number(e.target.value) : undefined }))} />
        <input type="text" placeholder="OT name" value={state.ot_name ?? ''} onChange={e => setState(s => ({ ...s, ot_name: e.target.value }))} />
        <input type="number" placeholder="OT TID" value={state.ot_tid ?? ''} onChange={e => setState(s => ({ ...s, ot_tid: e.target.value ? Number(e.target.value) : undefined }))} />
        <input type="number" placeholder="OT SID" value={state.ot_sid ?? ''} onChange={e => setState(s => ({ ...s, ot_sid: e.target.value ? Number(e.target.value) : undefined }))} />
      </Group>

      {/* Battle */}
      <Group title="Battle" open={openGroups.battle} onToggle={() => setOpenGroups(g => ({ ...g, battle: !g.battle }))}>
        <input type="text" placeholder="Nature" value={state.nature ?? ''} onChange={e => setState(s => ({ ...s, nature: e.target.value }))} />
        <input type="text" placeholder="Ability" value={state.ability ?? ''} onChange={e => setState(s => ({ ...s, ability: e.target.value }))} />
        <label><input type="checkbox" checked={!!state.is_hidden_ability} onChange={e => setState(s => ({ ...s, is_hidden_ability: e.target.checked }))} /> Hidden ability</label>
        <input type="text" placeholder="Held item" value={state.held_item ?? ''} onChange={e => setState(s => ({ ...s, held_item: e.target.value }))} />
        {/* IVs/EVs/moves: render simple inline editors; styling is the consumer's job. */}
      </Group>

      {/* Cosmetic / game-specific */}
      <Group title="Cosmetic / game-specific" open={openGroups.cosmetic} onToggle={() => setOpenGroups(g => ({ ...g, cosmetic: !g.cosmetic }))}>
        {showTera && (
          <select value={state.tera_type ?? ''} onChange={e => setState(s => ({ ...s, tera_type: e.target.value || undefined }))}>
            <option value="">Tera type…</option>
            {teraTypes.map(t => <option key={t.key} value={t.key}>{t.name}</option>)}
          </select>
        )}
        {showAlpha && (
          <label><input type="checkbox" checked={!!state.is_alpha} onChange={e => setState(s => ({ ...s, is_alpha: e.target.checked }))} /> Alpha</label>
        )}
        {showIsMega && (
          <label><input type="checkbox" checked={!!state.is_mega} onChange={e => setState(s => ({ ...s, is_mega: e.target.checked }))} /> Captured as Mega</label>
        )}
        {showMarks && marks.length > 0 && (
          <div>
            <div className="text-sm font-semibold">Marks</div>
            {marks.map(m => (
              <label key={m.name} className="block">
                <input type="checkbox"
                       checked={state.marks?.includes(m.name) ?? false}
                       onChange={e => setState(s => ({
                         ...s,
                         marks: e.target.checked
                           ? [...(s.marks ?? []), m.name]
                           : (s.marks ?? []).filter(n => n !== m.name),
                       }))} />
                {m.name} <span className="text-muted-foreground/60 text-xs">({m.category})</span>
              </label>
            ))}
          </div>
        )}
        <textarea placeholder="Notes" value={state.notes ?? ''} onChange={e => setState(s => ({ ...s, notes: e.target.value }))} />
      </Group>

      <div className="flex gap-2">
        <button type="submit" disabled={submitting || !state.species_id || !state.origin_game}>Save</button>
        <button type="button" onClick={onCancel}>Cancel</button>
      </div>
    </form>
  );
}

function Group({ title, open, onToggle, children }: { title: string; open: boolean; onToggle: () => void; children: React.ReactNode }) {
  return (
    <div className="border rounded-md">
      <button type="button" onClick={onToggle} className="w-full text-left px-3 py-2 font-semibold">
        {open ? '▼' : '▶'} {title}
      </button>
      {open && <div className="px-3 py-2 space-y-2">{children}</div>}
    </div>
  );
}
```

```typescript
import { useEffect, useState } from 'react';

interface TeraType { key: string; name: string; color: string; }

interface Props {
  speciesId: number;
  /** All entries the user owns for this species (any form, any origin). */
  entries: Array<{ tera_type?: string | null }>;
}

export function TeraLens({ speciesId, entries }: Props) {
  const [palette, setPalette] = useState<Record<string, string>>({});
  useEffect(() => {
    fetch('/api/reference/tera-types').then(r => r.json()).then((rows: TeraType[]) => {
      setPalette(Object.fromEntries(rows.map(t => [t.key, t.color])));
    }).catch(() => {});
  }, []);

  // Find the most-recent / most-distinct Tera type the user has. For the lens,
  // collapse to a single dominant color: first non-null tera_type wins.
  const teraType = entries.map(e => e.tera_type).find(Boolean) ?? null;
  const color = teraType ? palette[teraType] ?? '#444' : 'transparent';

  return (
    <div className="absolute inset-0 rounded-md pointer-events-none"
         style={{ boxShadow: teraType ? `inset 0 0 0 3px ${color}` : 'none' }}
         title={teraType ? `Tera ${teraType}` : 'No Tera entry'} />
  );
}
```
